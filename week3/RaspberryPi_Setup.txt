Installing Noble and Bleno with Debian Jessie on Raspberry Pi

https://www.raspberrypi.org/downloads/raspbian/
install
configure
apt-get update
apt-get dist-upgrade
sudo apt-get install bluetooth bluez libbluetooth-dev libudev-dev

curl -O https://nodejs.org/dist/v4.3.0/node-v4.3.0-linux-armv7l.tar.xz
tar xf node-v4.3.0-linux-armv7l.tar.xz
sudo mv node-v4.3.0-linux-armv7l /usr/local/
sudo ln -s /usr/local/node-v4.3.0-linux-armv7l/bin/node /usr/local/bin/
sudo ln -s /usr/local/node-v4.3.0-linux-armv7l/bin/npm /usr/local/bin/
